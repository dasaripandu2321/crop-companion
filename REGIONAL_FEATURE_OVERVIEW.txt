   ├── State-specific schemes

💻 USAGE EXAMPLE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

import { useState } from "react";
import { RegionalSummary } from "@/components/RegionalSummary";
import { RegionSelector } from "@/components/RegionSelector";

function PredictionResults({ prediction, soilData }) {
  const [region, setRegion] = useState("Punjab");
  
  return (
    <div>
      <h2>Predicted: {prediction.crop}</h2>
      
      <RegionSelector value={region} onChange={setRegion} />
      
      <RegionalSummary
        region={region}
        crop={prediction.crop}
        soilData={soilData}
      />
    </div>
  );
}

🐛 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Backend won't start
└── Check Python 3.8+, install dependencies, verify port 5000

Frontend can't connect
└── Ensure backend running, check CORS, verify API URL

Always shows mock data
└── Check .env has GEMINI_API_KEY, verify key is valid

Components not rendering
└── Verify imports, check tsconfig.json alias, install shadcn/ui
